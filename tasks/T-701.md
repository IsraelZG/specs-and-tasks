 * banco fГ­sico **isolado** do grafo. NГѓO toca em `nodes`/`edges` (T-106).

### Schema v1 (DDL) вЂ” banco `device_state.db` (isolado)

```sql
-- device_state.db вЂ” categoria Local+Persistente (caderno-3-sdk/01 В§4.1).
-- Banco FГЌSICO ISOLADO (OPFS no browser / arquivo no Node), separado do grafo (T-106).

CREATE TABLE IF NOT EXISTS drafts (
  id          TEXT    NOT NULL PRIMARY KEY,
  payload     BLOB    NOT NULL,
  updated_at  INTEGER NOT NULL
);

CREATE TABLE IF NOT EXISTS prefetch_cache (
  key         TEXT    NOT NULL PRIMARY KEY,
  blob        BLOB    NOT NULL,
  fetched_at  INTEGER NOT NULL,
  ttl_ms      INTEGER                       -- NULL = sem expiraГ§ГЈo
);

CREATE TABLE IF NOT EXISTS ui_preferences (
  key         TEXT    NOT NULL PRIMARY KEY,
  value       TEXT    NOT NULL,
  updated_at  INTEGER NOT NULL
);

CREATE TABLE IF NOT EXISTS peer_history (
  peer_id        TEXT    NOT NULL PRIMARY KEY,
  last_multiaddr TEXT,
  last_seen_at   INTEGER NOT NULL,
  success_count  INTEGER NOT NULL DEFAULT 0,
  failure_count  INTEGER NOT NULL DEFAULT 0
);

CREATE INDEX IF NOT EXISTS idx_peer_history_seen ON peer_history(last_seen_at);
```
