```ts
// --- packages/core/src/deviceState.ts

import type { StoragePort, ClockPort, PeerId } from '@plataforma/protocol'; // T-004 §1, T-105

/** Rascunho não-publicado (Local+Persistente). */
export interface DraftRecord {
  readonly id: string;            // ULID/entidade do rascunho
  readonly payload: Uint8Array;   // conteúdo serializado (MessagePack — opaco aqui)
  readonly updatedAt: number;     // epoch ms; carimbado via ClockPort (LWW será por HLC em T-703)
}

/** Entrada do cache de prefetch. */
export interface PrefetchEntry {
  readonly key: string;           // chave de cache (ex.: content/asset id)
  readonly blob: Uint8Array;
  readonly fetchedAt: number;     // epoch ms (carimbado via ClockPort)
  readonly ttlMs?: number;        // opcional; ausente = sem expiração
}

/** Entrada do histórico de peers (para warm resume — caderno-5 §2.4.2). */
export interface PeerHistoryEntry {
  readonly peerId: PeerId;
  readonly lastMultiaddr?: string; // hint de endereço para reconexão
  readonly lastSeenAt: number;     // epoch ms
  readonly successCount: number;
  readonly failureCount: number;
}

/**
 * Repositório tipado sobre o banco `device_state.db`. Reusa o `StoragePort` (T-004) apontado a um
 * banco físico **isolado** do grafo. NÃO toca em `nodes`/`edges` (T-106).
 *
 * @remarks Carimbos temporais (`updatedAt`/`fetchedAt`/`lastSeenAt`) vêm de `ClockPort.now()`
 *   (testável com VirtualClock T-003). T-703 poderá migrar a ordenação para HLC no merge.
 */
export class DeviceStateStore {
  constructor(storage: StoragePort, clock: ClockPort);

  /** Roda as migrations v1 (DDL das 4 tabelas). Idempotente. */
  init(): Promise<void>;

  // --- rascunhos ---
  putDraft(id: string, payload: Uint8Array): Promise<void>;  // upsert; carimba updatedAt
  getDraft(id: string): Promise<DraftRecord | undefined>;
  deleteDraft(id: string): Promise<void>;
  listDrafts(): Promise<DraftRecord[]>;

  // --- cache de prefetch ---
  putPrefetch(key: string, blob: Uint8Array, ttlMs?: number): Promise<void>; // upsert; carimba fetchedAt
  /** Retorna a entrada; se expirada (`fetchedAt + ttlMs < now`), retorna `undefined` E remove a linha (lazy expiry). */
  getPrefetch(key: string): Promise<PrefetchEntry | undefined>;

  // --- preferências de UI ---
  setPreference(key: string, value: string): Promise<void>;  // upsert (LWW por updatedAt)
  getPreference(key: string): Promise<string | undefined>;

  // --- histórico de peers ---
  /** Upsert: carimba lastSeenAt; incrementa success/failure conforme `outcome`; atualiza multiaddr se dado. */
  recordPeer(peerId: PeerId, opts?: { lastMultiaddr?: string; outcome?: 'success' | 'failure' }): Promise<void>;
  getPeer(peerId: PeerId): Promise<PeerHistoryEntry | undefined>;
  /** Os `limit` peers mais recentes (ordem `last_seen_at` desc) — para warm resume. */
  recentPeers(limit: number): Promise<PeerHistoryEntry[]>;
}
```

### Schema v1 (DDL) — banco `device_state.db` (isolado)

```sql
-- device_state.db — categoria Local+Persistente (caderno-3-sdk/01 §4.1).
-- Banco FÍSICO ISOLADO (OPFS no browser / arquivo no Node), separado do grafo (T-106).

CREATE TABLE IF NOT EXISTS drafts (
  id          TEXT    NOT NULL PRIMARY KEY,
  payload     BLOB    NOT NULL,
  updated_at  INTEGER NOT NULL
);

CREATE TABLE IF NOT EXISTS prefetch_cache (
  key         TEXT    NOT NULL PRIMARY KEY,
  blob        BLOB    NOT NULL,
  fetched_at  INTEGER NOT NULL,
  ttl_ms      INTEGER                       -- NULL = sem expiração
);

CREATE TABLE IF NOT EXISTS ui_preferences (
  key         TEXT    NOT NULL PRIMARY KEY,
  value       TEXT    NOT NULL,
  updated_at  INTEGER NOT NULL
);

CREATE TABLE IF NOT EXISTS peer_history (
  peer_id        TEXT    NOT NULL PRIMARY KEY,
  last_multiaddr TEXT,
  last_seen_at   INTEGER NOT NULL,
  success_count  INTEGER NOT NULL DEFAULT 0,
  failure_count  INTEGER NOT NULL DEFAULT 0
);

CREATE INDEX IF NOT EXISTS idx_peer_history_seen ON peer_history(last_seen_at);
```

> **Extensibilidade:** as colunas v1 cobrem o uso atual; tasks consumidoras (T-406 etc.) podem
> adicionar colunas via nova migration versionada (`StoragePort.migrate`), sem quebrar v1.

Casos de teste (numerados, implementar 1:1):
1. **init cria as 4 tabelas:** `init()` → `drafts`, `prefetch_cache`, `ui_preferences`, `peer_history` existem (consultar `sqlite_master`). `init()` 2× é idempotente (sem erro).
2. **Draft round-trip:** `putDraft('d1', bytes)` → `getDraft('d1')` devolve `payload` igual byte-a-byte e `updatedAt === clock.now()`.
3. **Draft upsert:** `putDraft('d1', a)`; avança clock; `putDraft('d1', b)` → `getDraft('d1').payload === b` e `updatedAt` atualizado.
4. **deleteDraft:** após `deleteDraft('d1')`, `getDraft('d1') === undefined`.
5. **listDrafts:** 3 drafts inseridos → `listDrafts()` retorna os 3.
6. **Prefetch round-trip:** `putPrefetch('k', blob, 1000)` → `getPrefetch('k')` devolve `blob` + `ttlMs === 1000`.
7. **Prefetch lazy expiry:** `putPrefetch('k', blob, 1000)`; `clock.advance(1001)` → `getPrefetch('k') === undefined` **e** a linha foi removida (segundo `getPrefetch` continua `undefined`; verificar no DB).
8. **Prefetch sem TTL não expira:** `putPrefetch('k', blob)` (ttl ausente); `clock.advance(10_000_000)` → `getPrefetch('k')` ainda devolve o blob.
9. **Preference round-trip + LWW:** `setPreference('theme','dark')` → `getPreference('theme') === 'dark'`; `setPreference('theme','light')` sobrescreve → `'light'`.
10. **Preference inexistente:** `getPreference('nope') === undefined`.
11. **recordPeer upsert + outcome:** `recordPeer('p1', { outcome:'success' })` → `getPeer('p1').successCount === 1`; segundo `recordPeer('p1', { outcome:'failure', lastMultiaddr:'/ip4/...' })` → `successCount 1`, `failureCount 1`, `lastMultiaddr` atualizado, `lastSeenAt` recarimbado.
12. **recentPeers ordena e limita:** inserir p1,p2,p3 em instantes crescentes; `recentPeers(2)` → `[p3, p2]` (mais recentes primeiro).
13. **getPeer inexistente:** `getPeer('zzz') === undefined`.
14. **Isolamento do grafo:** após todas as operações, `sqlite_master` **não** contém `nodes` nem `edges` (este banco é isolado — T-106 é outro DB).
15. **Tudo sob `StoragePort` injetado:** nenhuma abertura direta de arquivo/OPFS no código de produção; o store opera só via o `StoragePort` recebido no construtor.
