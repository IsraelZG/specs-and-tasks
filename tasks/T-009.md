### Contratos exatos
```ts
// --- packages/core/src/ports/ControlPort.ts ---
import { PeerId } from '@plataforma/protocol'; // T-004/T-105

export interface ControlCommand {
  corrId: string;
  cmd: string;           // 'drop_connection' | 'partition' | 'sync_now' | 'dump_state'
  payload: unknown;
}

export interface ControlEvent {
  type: string;          // 'peer_connected' | 'peer_disconnected' | 'sync_complete' | 'error'
  peerId?: PeerId;
  data: unknown;
  timestamp: number;
}

export type ControlEventHandler = (event: ControlEvent) => void;

export interface ControlPort {
  /** Executa comando de controle, retorna resultado. */
  execute(command: ControlCommand): Promise<unknown>;

  /** Registra listener de eventos de telemetria. */
  onEvent(handler: ControlEventHandler): void;

  /** Força desconexão de peer (sem matar processo). */
  dropConnection(peerId: PeerId): Promise<void>;

  /** Particiona rede localmente (para testes). */
  setPartition(groups: PeerId[][]): Promise<void>;

  /** Restaura conectividade total. */
  healPartition(): Promise<void>;
}
```

```ts
// --- packages/client-sdk/src/WebSocketControlClient.ts ---
import { ControlPort } from '@plataforma/core';

export class WebSocketControlClient {
  constructor(url: string, token: string);

  /** Conecta ao Hub de QA e faz handshake (client_id + token). */
  connect(): Promise<void>;

  /** Vincula ControlPort local para despachar comandos recebidos. */
  bind(controlPort: ControlPort): void;

  /** Envia evento de telemetria ao Hub (push). */
  sendTelemetry(event: ControlEvent): void;

  /** Desconecta. */
  close(): void;
}
```

## 4. Estratégia de Testes
- [x] **Framework:** Vitest (Node puro).
- [x] **Ambiente:** `pnpm --filter @plataforma/client-sdk test`.

Casos de teste:
1. `WebSocketControlClient.connect` → handshake bem-sucedido com servidor `ws` in-memory.
2. Servidor envia `{corrId, cmd:'drop_connection', payload:{peerId:'A'}}` → `ControlPort.dropConnection('A')` chamado.
3. Cliente responde com mesmo `corrId` após executar comando.
4. `sendTelemetry` → servidor recebe evento estruturado.
5. Conexão cai → reconexão automática (até 3 tentativas).
6. Handshake com token inválido → rejeitado.
