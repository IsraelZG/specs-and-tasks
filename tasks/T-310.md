## 1. Objetivo

### Contratos exatos
```ts
// --- packages/transport/src/transportMatrix.ts 
---

export enum TransportTarget {
  NODES_EDGES = 'nodes_edges',           // RBSR — grafo durável
  PENDING_CHANGES = 'pending_changes',   // efêmero — canal Automerge/WebRTC
  DEVICE_STATE_DB = 'device_state_db',   // private swarm — device_state.db
  RAM = 'ram',                           // volátil — SwarmRegistry, cache de prefetch
}

export interface TransportHint {
  entity_type: string;          // tipo de entidade (ex: "CONTENT", "SPECIFICATION")
  target: TransportTarget;
  priority?: number;            // 0=mais alta
  condition?: string;           // expressão opcional de condição (ex: "scope=private")
}

export type TransportMatrix = TransportHint[];

/** Avalia as hints em ordem de prioridade e retorna o target para uma entidade. */
export function evaluateTransportHints(
  hints: TransportMatrix,
  entityType: string,
  scope?: string,
): TransportTarget;

/** Tabela de verdade: todas as combinações entity_type × condition → target. */
export function validateMatrix(hints: TransportMatrix): { valid: boolean; errors: string[] };

/** Matriz default do sistema. */
export const DEFAULT_TRANSPORT_MATRIX: TransportMatrix;
```
- `evaluateTransportHints`: percorre hints ordenadas por priority; retorna o primeiro match.
- `validateMatrix`: valida que toda combinação de entity_type + condition tem target único; detecta ambiguidades.
- `DEFAULT_TRANSPORT_MATRIX`: SPECIFICATION → RAM, CONTENT → NODES_EDGES, pending → PENDING_CHANGES, device_state → DEVICE_STATE_DB.
- Tipo errado não compila/roteia (type-safe via union discriminada).
