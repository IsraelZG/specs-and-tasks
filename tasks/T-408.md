### Contratos exatos
```ts
// --- apps/system-peer/src/tracker.ts ---
import { PeerId } from '@plataforma/protocol';
import { SignedUcan } from '@plataforma/core';

export interface TrackerAnnounce {
  infoHash: string;              // InfoHash do conteГєdo (hex)
  peerId: PeerId;
  uploaded: number;              // bytes uploaded
  downloaded: number;            // bytes downloaded
  left: number;                  // bytes restantes
  event: 'started' | 'stopped' | 'completed' | 'update';
  ucan: SignedUcan;              // UCAN autorizando acesso ao swarm
}

export interface TrackerPeer {
  peerId: PeerId;
  ip: string;
  port: number;
  left: number;
  lastSeenAt: number;
}

export interface TrackerScrape {
  infoHash: string;
  ucan: SignedUcan;
}

export interface TrackerStats {
  complete: number;              // seeders
  incomplete: number;            // leechers
  downloaded: number;            // vezes que foi completado
}

export interface TrackerService {
  /** Anuncia presenГ§a no swarm. */
  announce(req: TrackerAnnounce): Promise<{ interval: number; peers: TrackerPeer[] }>;

  /** Consulta estatГ­sticas de um swarm. */
  scrape(req: TrackerScrape): Promise<{ files: Record<string, TrackerStats> }>;

  /** Valida UCAN antes de permitir announce/scrape. */
  validateUcan(ucan: SignedUcan, infoHash: string): Promise<boolean>;

  /** Remove peers inativos (TTL 30min sem announce). */
  purgeStale(): number;
}
```
- `announce`: registra peer como seeder/leecher; valida UCAN; retorna lista de peers no swarm.
- `scrape`: retorna estatГ­sticas (completed, incomplete, downloaded).
- `validateUcan`: verifica assinatura + expiraГ§ГЈo + escopo inclui o infoHash.
- `purgeStale`: remove peers sem announce hГЎ > 30min.
- Endpoints: `GET /tracker/announce` e `GET /tracker/scrape` via WSS.
