### Contratos exatos (assinaturas TS fixadas)

```ts
// --- packages/protocol/src/wireFormat.ts 
---

/** Versão corrente do protocolo. Derivado de caderno-2-protocol/05-wire-protocol.md §1. */
export const WIRE_VERSION = 0x01;

/** Tipos de frame — tabela do caderno-2-protocol/05-wire-protocol.md §1. */
export enum FrameType {
  RBSR_ROOT          = 0x01,
  RBSR_RANGE         = 0x02,
  RBSR_REQUEST_NODES = 0x03,
  RBSR_NODES         = 0x04,
  EPHEMERAL          = 0x05,
  HEARTBEAT_PING     = 0x06,
  HEARTBEAT_PONG     = 0x07,
  IDENTITY_CATCHUP   = 0x08,
  KEY_REQUEST        = 0x09,
  KEY_RESPONSE       = 0x0A,
  CONTROL            = 0x0B,
}

/** Teto de tamanho de frame (1 MiB). Caderno §1. */
export const MAX_FRAME_LENGTH = 1_048_576;

/** Frame decodificado. */
export interface WireFrame {
  version: number;
  frameType: FrameType;
  payload: unknown; // MessagePack-decoded (any JSON-compatible)
}

/** Resultado de decode: frame válido ou rejeição/quarentena. */
export type DecodeResult =
  | { ok: true; frame: WireFrame }
  | { ok: false; reason: 'too_large' | 'bad_length' | 'unknown_version' | 'parse_error'; retentionState?: number; raw?: Uint8Array };

/** Codifica payload MessagePack em frame binário. */
export function encodeFrame(frameType: FrameType, payload: unknown): Uint8Array;

/** Decodifica frame binário. Nunca lança — versão futura retorna {ok:false} com retentionState. */
export function decodeFrame(buffer: Uint8Array): DecodeResult;

/** Lê frame de stream incremental (buffer + socket). Retorna {frame, consumed} ou null se incompleto. */
export function tryReadFrame(buffer: Uint8Array): { frame: DecodeResult; consumed: number } | null;
```

### Regras de quarentena (derivado do caderno §3)
- Versão > WIRE_VERSION + frame volátil (`EPHEMERAL`, `HEARTBEAT_*`, `CONTROL`) → descartar (`{ok: false, reason: 'unknown_version'}`).
- Versão > WIRE_VERSION + frame auditável → `{ok: false, reason: 'unknown_version', retentionState: 3, raw}`.
- Versão < 0x01 → `{ok: false, reason: 'unknown_version'}`.
- LENGTH > MAX_FRAME_LENGTH → `{ok: false, reason: 'too_large'}`.
