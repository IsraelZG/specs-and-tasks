## 3. Escopo de Arquivos (Inputs e Outputs)
- **[UPDATE]** `tools/scripts/manage-task.mjs` — manter o parse de args
  (`<action> <taskId> <agentName> [mensagem...]`) e substituir TODA a lógica de
  escrita/regex/rebuild por delegação ao serviço compilado:
  ```js
  import path from 'path';
  import { fileURLToPath, pathToFileURL } from 'url';

  const __dirname = path.dirname(fileURLToPath(import.meta.url));
  const repoRoot = path.resolve(__dirname, '../..');           // tools/scripts -> raiz
  const distPath = path.resolve(repoRoot, 'apps/nexus-backend/dist/services/task.service.js');

  let TaskService;
  try {
    ({ TaskService } = await import(pathToFileURL(distPath).href));
  } catch {
    console.error('Backend não compilado. Rode: pnpm --filter nexus-backend build');
    process.exit(1);
  }

  const svc = new TaskService({ rootDir: repoRoot });
  try {
    const rec = svc.transition(taskId, action, agentName, message); // action: start|pause|finish|...
    console.log(`✅ Tarefa ${taskId} atualizada. Status: ${rec.frontmatter.status}. Log adicionado.`);
  } catch (err) {
    console.error(`❌ ${err.message}`);
    process.exit(1);
  }
  ```
  (O wrapper passa a aceitar naturalmente também `approve|request_changes|block|unblock`,
  pois delega ao `transition`. `start|pause|finish` continuam funcionando igual.)
