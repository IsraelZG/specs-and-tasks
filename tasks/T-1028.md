```ts
// --- task.types.ts ---

export type TaskStatus =
  | 'draft'                    // LEGADO: alias de draft:placeholder (tolerado até a migração T-1030 rodar)
  | 'draft:placeholder'
  | 'draft:triaged'
  | 'draft:pending_decision'
  | 'draft:hardened'
  | 'draft:decomposed'
  | 'ready'
  | 'in_progress'
  | 'review'
  | 'in_review'
  | 'rework'
  | 'done'
  | 'blocked';

export type TaskAction =
  | 'triage' | 'harden' | 'decide' | 'block_decision' | 'decompose'  // endurecimento (draft ladder)
  | 'promote' | 'start' | 'pause' | 'finish'
  | 'claim' | 'approve' | 'request_changes'
  | 'block' | 'unblock';

/** Base do status: 'draft:hardened' → 'draft'. Usado pelos consumidores (T-1030) p/ agrupar draft:*. */
export function baseStatus(s: TaskStatus | string): string {
  return s.includes(':') ? s.split(':')[0] : s;
}

export const TRANSITIONS: Record<TaskAction, TransitionRule> = {
  // --- endurecimento (draft ladder) ---
  triage:         { from: ['draft', 'draft:placeholder'],                        to: 'draft:triaged',          logLabel: '[Triado]' },
  harden:         { from: ['draft:triaged'],                                     to: 'draft:hardened',         logLabel: '[Endurecido]' },
  decide:         { from: ['draft:pending_decision'],                            to: 'draft:hardened',         logLabel: '[Decidido]' },
  block_decision: { from: ['draft', 'draft:placeholder', 'draft:triaged'],       to: 'draft:pending_decision', logLabel: '[Decisão pendente]' },
  decompose:      { from: '*',                                                   to: 'draft:decomposed',       logLabel: '[Decomposto]' },
  // --- lifecycle ---
  promote:        { from: ['draft', 'draft:placeholder', 'draft:triaged', 'draft:hardened'], to: 'ready',      logLabel: '[Promovida p/ ready]' },
  start:          { from: ['ready', 'rework'],                                   to: 'in_progress',            logLabel: '[Iniciado]' },
  pause:          { from: ['in_progress'],                                       to: 'in_progress',            logLabel: '[Pausado/Handoff]' },
  finish:         { from: ['in_progress'],                                       to: 'review',                 logLabel: '[Finalizado]' },
  claim:          { from: ['review'],                                            to: 'in_review',              logLabel: '[Em revisão]' },
  approve:        { from: ['review', 'in_review'],                               to: 'done',                   logLabel: '[Aprovado]' },
  request_changes:{ from: ['review', 'in_review'],                               to: 'rework',                 logLabel: '[Requer Refatoração]' },
  block:          { from: '*',                                                   to: 'blocked',                logLabel: '[Bloqueado]' },
  unblock:        { from: ['blocked'],                                           to: 'ready',                  logLabel: '[Desbloqueado]' },
};
```

**Decisões de desenho já fechadas (não reabrir):**
- **`start.from` NÃO inclui mais `draft`/`draft:*`.** Com auto-promote (T-1029) uma task só chega a `ready` já endurecida; começar um draft direto deixa de ser permitido. Isso, de brinde, barra `start` num `draft:decomposed` (pai container). A trava explícita por `subtasks` fica na T-1029 (belt-and-suspenders).
- **`claim` é gated ao papel do reviewer**, igual `approve`/`request_changes` — impede que o worker "reivindique" a própria revisão.
- **Verbos de endurecimento (`triage`/`harden`/`decide`/`block_decision`/`decompose`) NÃO são gated a papel.** Endurecer não tem conflito de auto-aprovação; qualquer modelo pode rodar (mesmo tratamento de `start`/`finish` hoje). NÃO adicione gate de arquiteto.
- **`'draft'` permanece no union como alias legado** de `draft:placeholder` — durante a janela entre esta task e a migração (T-1030) os arquivos ainda dizem `draft`; `triage`/`promote`/`block_decision` aceitam ambos. Depois da migração nenhum arquivo diz `draft` puro, mas o tipo tolera.
- **`approve`/`request_changes` aceitam `from: ['review','in_review']`** — reviewer manual/legado que aprova sem `claim` continua funcionando; o lock efetivo vem do orquestrador tratar `in_review` como ocupado (T-1030 `nextAction`).
