### Contratos exatos
```ts
// --- packages/transport/src/promotion/Engine.ts 
---
import { PeerId } from '@plataforma/protocol'; // T-004

export enum PromotionState {
  RELAY_ONLY    = 'relay_only',
  PROMOTING      = 'promoting',
  DIRECT         = 'direct',
  FAILED         = 'failed',
}

export interface PromotionEntry {
  peerId: PeerId;
  state: PromotionState;
  attempts: number;
  lastAttemptAt: number;
}

export class ConnectionPromotionEngine {
  constructor(maxAttempts?: number); // default 3

  /** Tenta promover conexão relay → direta. Retorna novo estado. */
  tryPromote(peerId: PeerId, natType: NatType): Promise<PromotionState>;

  /** Estado atual da promoção para peer. */
  getState(peerId: PeerId): PromotionState;

  /** Cancela tentativas de promoção. */
  cancel(peerId: PeerId): void;

  /** Número de conexões diretas ativas. */
  get directCount(): number;
}
```
- NAT `Symmetric`: nunca promove → `FAILED` na primeira tentativa.
- NAT `Cone`: promove em 100% dos casos (sem bloqueio de porta).
- Make-before-break: conexão relay só é fechada após WebRTC confirmado.
