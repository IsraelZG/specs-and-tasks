## 1. Objetivo
Implementar o motor de linhagem (Layer 2) no pacote `@plataforma/core`: navegação em DAG via arestas `MUTATES`, validação de cadeia de hash, detecção de ciclos, e seleção de `head` por HLC máximo
*(derivado de `docs/visao-arquitetural.md` §5 "Imutabilidade Dupla" — Layer 2: aresta MUTATES referencia hash da versão anterior; `docs/conceitos/entity-id.md` — entity_id herda do nó-raiz; `docs/conceitos/hlc.md` — head = nó de maior HLC na linhagem)*.

### Contratos exatos (assinaturas TS fixadas)

```ts
// --- packages/core/src/lineage.ts ---
import { SignedNode, SignedEdge } from './signature.js'; // T-107
import { ULID, EntityId } from './ulid.js'; // T-102
import { HLCTimestamp } from './hlc.js'; // T-103
import { StoragePort } from '@plataforma/protocol'; // T-004

/** Hash SHA-256 de um nó (32 bytes). Usado na aresta MUTATES para referenciar o pai. */
export type NodeHash = Uint8Array;

/** Calcula SHA-256 do canonicalize(nó) — identifica unicamente o conteúdo. */
export function hashNode(node: SignedNode): NodeHash;

// === Operações de linhagem ===

/** Insere nó no grafo com validação Layer 2:
 *  1. Se há pai (aresta MUTATES de entrada), valida que:
 *     - parentHash confere com hashNode(pai)
 *     - HLC(filho) > HLC(pai) (invariante de pai — hlc.md)
 *     - entity_id do filho = entity_id do pai (herança — entity-id.md)
 *  2. Se é nó-raiz (sem pai), entity_id = id.
 *  3. Detecta e rejeita ciclos.
 *  4. Persiste nó + aresta MUTATES no storage.
 *  5. Atualiza entity_heads (insere ou UPDATE se HLC maior).
 */
export async function insertNode(
  storage: StoragePort,
  node: SignedNode,
  parentId?: ULID,
): Promise<void>;

/** Navega a linhagem de uma entidade do head à raiz (ordem reversa de HLC). */
export async function getLineage(storage: StoragePort, entityId: EntityId): Promise<SignedNode[]>;

/** Retorna o head atual de uma entidade (nó com maior HLC na entity_heads). */
export async function getHead(storage: StoragePort, entityId: EntityId): Promise<SignedNode | null>;

/** Verifica se a cadeia de hash é íntegra da raiz ao head. */
export async function validateChain(storage: StoragePort, entityId: EntityId): Promise<boolean>;

/** Detecta se há fork (duas arestas MUTATES ativas com mesmo source_id). */
export async function detectFork(storage: StoragePort, sourceId: ULID): Promise<boolean>;
```

### Tabelas adicionais (migration v2 do schema T-106)
```sql
-- entity_heads: um registro por linhagem, apontando pro nó mais recente
CREATE TABLE IF NOT EXISTS entity_heads (
  entity_id  TEXT NOT NULL PRIMARY KEY,  -- EntityId (ULID do nó-raiz)
  head_id    TEXT NOT NULL REFERENCES nodes(id),
  head_hlc   INTEGER NOT NULL           -- HLCTimestamp empacotado (bigint como INTEGER)
);

-- Gatilho: ao inserir nó, atualiza entity_heads se HLC maior
CREATE TRIGGER IF NOT EXISTS trg_nodes_insert_entity_head
AFTER INSERT ON nodes
BEGIN
  INSERT INTO entity_heads (entity_id, head_id, head_hlc)
  VALUES (NEW.entity_id, NEW.id, NEW.hlc)
  ON CONFLICT(entity_id) DO UPDATE
  SET head_id = NEW.id, head_hlc = NEW.hlc
  WHERE NEW.hlc > head_hlc;
END;
```
