### Contratos exatos
```ts
// --- packages/core/src/sync/mergeStrategies.ts ---
import { EntityId, ULID } from '@plataforma/protocol'; // T-102

export enum MergeClass {
  PREFERENCE = 'preference',     // LWW por HLC
  DRAFT = 'draft',               // Automerge (CRDT)
  PREFETCH_CACHE = 'prefetch_cache', // union-based
}

export interface MergeInput {
  localValue: unknown;
  remoteValue: unknown;
  localHlc: number;
  remoteHlc: number;
  mergeClass: MergeClass;
}

export interface MergeResult {
  resolved: unknown;
  strategy: MergeClass;
  conflict: boolean;
  conflictResolution: 'lww' | 'automerge' | 'union' | 'none';
}

export interface MergeStrategies {
  /** LWW: valor com maior HLC vence. */
  lwwMerge(input: MergeInput): MergeResult;

  /** Automerge: merge CRDT para rascunhos. */
  automergeMerge(input: MergeInput): MergeResult;

  /** Union: união de caches de prefetch (sem sobrescrita). */
  unionMerge(input: MergeInput): MergeResult;

  /** Seleciona estratégia baseada na classe declarada. */
  merge(input: MergeInput): MergeResult;

  /** Registra classe customizada (extensível). */
  registerClass(mergeClass: MergeClass, strategy: (input: MergeInput) => MergeResult): void;
}
```
- `lwwMerge`: `localHlc > remoteHlc` ? local : remote (Last Writer Wins).
- `automergeMerge`: usa Automerge Repo para merge de rascunhos (CRDT sem perda).
- `unionMerge`: união de conjuntos (cache de prefetch) — items do local + remoto sem duplicação.
- Dois devices offline editando rascunho → convergem sem perda via Automerge.
- União de caches verificada (items duplicados removidos).
