## 1. Objetivo

### Contratos
```tsx
// --- apps/bancada/src/tabs/MidiaTab.tsx 
---
import type { InfoHash, ChunkingMode } from '@plataforma/media'; // T-801/T-804 — fonte única (Decisão §6 #1)

/** Consulta o system-peer de verdade — sem mock (Decisão §6 #3). */
export interface P2PMonitorPort {
  getSeeders(infoHash: InfoHash): Promise<{ peerId: string; progress: number; speed: number }[]>;
  getLeecherCount(infoHash: InfoHash): Promise<number>;
  isWebSeedAvailable(infoHash: InfoHash): Promise<boolean>;
}

export interface FileUploadState {
  infoHash: InfoHash | null;
  fileName: string;
  size: number;
  cipherMode: ChunkingMode;
  progress: number;              // 0–100
  seeders: number;
  leechers: number;
  webSeedAvailable: boolean;
}

export interface MidiaTabProps {
  onUpload: (file: File, mode: ChunkingMode) => Promise<InfoHash>;
  onPlay: (infoHash: InfoHash) => void;
  /** Chama DELETE /webseed/:infoHash no system-peer real (T-804 killWebSeed) — não é simulação de UI (Decisão §6 #2). */
  onKillWebSeed: (infoHash: InfoHash) => Promise<void>;
  /** Injetado — consulta o system-peer de verdade (Decisão §6 #3). */
  p2pMonitor: P2PMonitorPort;
}

export function MidiaTab(props: MidiaTabProps): JSX.Element;
```
- Upload: input file + seletor de modo de cifra (`convergent` | `unique`).
- Progresso de upload com barra.
- Visualização: tabela de seeders (peerId, progress, speed) via `p2pMonitor.getSeeders` e fontes (WebTorrent, WebSeed) via `p2pMonitor.isWebSeedAvailable` — dados reais, sem mock (§6 #3).
- Player: `<video>` ou `<audio>` para o arquivo cifrado.
- Botão "Derrubar WebSeed" → chama `onKillWebSeed` (HTTP real `DELETE /webseed/:infoHash` no system-peer, T-804 — §6 #2), depois reconsulta `p2pMonitor` e reflete o resultado na tabela.
