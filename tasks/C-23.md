     ```typescript
     export function assertValidModelIdentity(
       actor: string,
       options?: IdentityGuardOptions,
     ): void {
       if (options?.bypass) return;
       const blockedActors = options?.blockedActors ?? DEFAULT_BLOCKED_ACTORS;
       const blockedRolePrefixes = options?.blockedRolePrefixes ?? DEFAULT_BLOCKED_ROLE_PREFIXES;

       // Extrai a parte do modelo se houver um padrão prefixo:modelo
       let modelName = actor;
       const parts = actor.split(":");
       if (parts.length > 1) {
         const possiblePrefix = parts[0];
         if (possiblePrefix && blockedRolePrefixes.includes(possiblePrefix)) {
           modelName = parts.slice(1).join(":");
         }
       }

       if (blockedActors.has(modelName)) {
         throw new GuardError(
           `Identidade inválida: "${actor}" tenta mascarar o harness "${modelName}"`,
         );
       }

       if (blockedActors.has(actor)) {
         throw new GuardError(
           `Identidade inválida: "${actor}" é um harness, não um modelo real`,
         );
       }

       for (const prefix of blockedRolePrefixes) {
         if (actor.startsWith(prefix)) {
           if (actor.length > prefix.length && actor[prefix.length] === ":") {
             continue;
           }
           throw new GuardError(
             `Identidade inválida: "${actor}" é um papel, não um modelo real`,
           );
         }
       }
     }
     ```
