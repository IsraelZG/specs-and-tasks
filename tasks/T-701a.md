## 1. Objetivo
Criar o schema oficial SQLite e as migrations versionadas para o banco de dados de estado local isolado (`device_state.db`) no pacote `@plataforma/core`, aplicando-as via `StoragePort.migrate()`.
*(derivado de `docs/caderno-3-sdk/01-sqlite-and-projections-schema.md` §4.1; `tasks/T-707.md` e `tasks/T-313c.md`)*.

### Contratos exatos (DDL e assinaturas TS fixadas)

```ts
// --- packages/core/src/deviceState/schema.ts ---
import type { StoragePort } from '@plataforma/protocol';

export interface DeviceStateMigration {
  version: number;
  sql: string;
}

export const DEVICE_STATE_MIGRATIONS: DeviceStateMigration[] = [
  {
    version: 1,
    sql: `
      CREATE TABLE IF NOT EXISTS drafts (
        id          TEXT    NOT NULL PRIMARY KEY,
        payload     BLOB    NOT NULL,
        updated_at  INTEGER NOT NULL
      );

      CREATE TABLE IF NOT EXISTS prefetch_cache (
        key         TEXT    NOT NULL PRIMARY KEY,
        blob        BLOB    NOT NULL,
        fetched_at  INTEGER NOT NULL,
        ttl_ms      INTEGER
      );

      CREATE TABLE IF NOT EXISTS ui_preferences (
        key         TEXT    NOT NULL PRIMARY KEY,
        value       TEXT    NOT NULL,
        updated_at  INTEGER NOT NULL
      );

      CREATE TABLE IF NOT EXISTS peer_history (
        peer_id        TEXT    NOT NULL PRIMARY KEY,
        last_multiaddr TEXT,
        last_seen_at   INTEGER NOT NULL,
        success_count  INTEGER NOT NULL DEFAULT 0,
        failure_count  INTEGER NOT NULL DEFAULT 0
      );

      CREATE INDEX IF NOT EXISTS idx_peer_history_seen ON peer_history(last_seen_at);
    `
  },
  {
    version: 2,
    sql: `
      CREATE TABLE IF NOT EXISTS blind_archives (
        archive_id     BLOB    NOT NULL PRIMARY KEY,
        blind_scope_id BLOB    NOT NULL,
        manifest       BLOB    NOT NULL,
        ciphertext     BLOB    NOT NULL,
        expires_at     INTEGER NOT NULL,
        created_at     INTEGER NOT NULL
      );
    `
  }
];

/** Aplica todas as migrations pendentes do device_state via StoragePort. Idempotente. */
export async function migrateDeviceStateSchema(storage: StoragePort): Promise<void>;
```
